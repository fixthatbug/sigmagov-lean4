\section{NPL Axioms}

\begin{lstlisting}
axiom ts_test_perfect_convergence :
  forall (nplP nplR : NPL),
    nplP.dimensions = nplR.dimensions ->
    cosineSimilarity nplP.embeddings nplR.embeddings >= tau_npl ->
    nplConverged nplP nplR tau_npl dimensionEpsilon

axiom ts_test_semantic_misalignment :
  forall (nplP nplR : NPL) (tau : Float) (epsilon : Dimension -> Float),
    cosineSimilarity nplP.embeddings nplR.embeddings < tau ->
    Not (nplConverged nplP nplR tau epsilon)

\section{Workflow Axioms}

\begin{lstlisting}
axiom par_comm_result :
  forall (w1 : Workflow a) (w2 : Workflow b) (p : Purpose),
    exists (swap : ParResult a b -> ParResult b a),
      Functor.map swap ((w1 ||| w2) p) = (w2 ||| w1) p
\end{lstlisting}

\section{Axiom Loader Axioms}

\begin{lstlisting}
axiom projection_preserves_dependencies :
  forall foundation ctx a,
    a in projectAxioms foundation ctx ->
    forall d in a.dependencies,
      d in (projectAxioms foundation ctx).map getId

axiom tool_addition_monotonic :
  forall foundation ctx1 ctx2,
    ctx1.tools <= ctx2.tools ->
    projectAxioms foundation ctx1 <= projectAxioms foundation ctx2
\end{lstlisting}
